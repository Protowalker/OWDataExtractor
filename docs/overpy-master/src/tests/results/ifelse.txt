variables {
    global:
        0: A
        1: B
        2: C
        3: D
        4: E
        5: F
}
rule ("simple if/else") {
    event {
        Ongoing - Global;
    }
    actions {
        If(Global.A);
            Global.A += 1;
        Else If(Global.B);
            Global.B += 1;
        Else;
            Global.C += 1;
        End;
        Global.D += 1;
    }
}

rule ("multiple elif") {
    event {
        Ongoing - Global;
    }
    actions {
        If(Global.A);
            Global.E += 1;
        Else If(Global.B);
            Global.E += 1;
        Else If(Global.C);
            Global.E += 1;
        Else If(Global.D);
            Global.E += 1;
    }
}

rule ("nested if") {
    event {
        Ongoing - Global;
    }
    actions {
        If(Global.A);
            If(Global.B);
                If(Global.C);
                    Global.D += 1;
                End;
            Else;
                Global.E += 1;
            End;
        End;
        Global.F += 1;
    }
}

rule ("if that goes to the end") {
    event {
        Ongoing - Global;
    }
    actions {
        If(Global.A);
            If(Global.B);
                Global.C += 1;
    }
}

rule ("special if with RULE_CONDITION") {
    event {
        Ongoing - Global;
    }
    actions {
        Abort If Condition Is True;
        Loop If Condition Is True;
        Loop If Condition Is True;
        If(Compare(Global.A, ==, 2));
            Set Global Variable(B, 3);
        Else;
            Abort If Condition Is False;
            Loop If Condition Is False;
            Set Global Variable(A, 1);
            Loop If Condition Is False;
            Set Global Variable(A, 2);
        End;
        Loop If Condition Is True;
    }
}

rule ("abort/loop/skip if") {
    event {
        Ongoing - Global;
    }
    actions {
        Skip If(Global.A, Global.B);
        Skip(Global.B);
        "pass"
        disabled Abort;
        Skip If(Global.A, 14);
        Skip(13);
        "pass"
        disabled Abort;
        If(Global.A);
            Loop;
            Global.B += 1;
        End;
        Loop If(Global.A);
        Loop;
        "pass"
        disabled Abort;
        If(Global.A);
            Loop;
        Else;
            Global.B += 1;
        End;
        //lbl_0:
        Abort If(Global.A);
        Abort;
        "pass"
        disabled Abort;
    }
}
